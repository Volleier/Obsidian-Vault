import google.generativeai as genai
import os

# Configure the Judge Model
genai.configure(api_key="YOUR_GEMINI_API_KEY")
model = genai.GenerativeModel('gemini-1.5-pro')

def load_ldr(folder, structure_id):
    path = f"{folder}/{structure_id}.ldr"
    if os.path.exists(path):
        with open(path, 'r') as f: return f.read()
    return "N/A"

def conduct_judgment(structure_id, prompt_text):
    # Load outputs from all three sources
    brick_ldr = load_ldr(f"Part_A_BrickGPT_Outputs/{structure_id}", structure_id)
    qwen_ldr = load_ldr("Part_B_Comparison/Qwen3_Outputs", structure_id)
    gemini_ldr = load_ldr("Part_B_Comparison/Gemini3_Outputs", structure_id)

    with open("judge_prompts.txt", "r") as f:
        system_instructions = f.read()

    query = f"""
    {system_instructions}
    
    Structure ID: {structure_id}
    Original Prompt: {prompt_text}
    
    [Model 1: BrickGPT LDR]
    {brick_ldr}
    
    [Model 2: Qwen3 LDR]
    {qwen_ldr}
    
    [Model 3: Gemini 3 LDR]
    {gemini_ldr}
    
    Provide your verdict.
    """
    
    response = model.generate_content(query)
    return response.text
